```
import sys
import requests
import json
import time

APP_ID = "cli_a97be1d34a781ceb"
APP_SECRET = "3HmPPOSa0g9nB1w9Vtftgd6aJaKFKMAx"
OPEN_ID = "ou_516f79447932bb772bae0ffc10bf9e46"

def push_text(text, max_retries=3):
    """
    Push a text message to Feishu user.
    Retries up to max_retries times on failure.
    """
    # Truncate to 3800 characters as Feishu limit
    text = text[:3800]

    for attempt in range(max_retries):
        try:
            # Get tenant access token
            res = requests.post(
                "https://open.feishu.cn/open-apis/auth/v3/tenant_access_token/internal",
                json={"app_id": APP_ID, "app_secret": APP_SECRET},
                timeout=10
            )
            res.raise_for_status()
            token = res.json()["tenant_access_token"]

            # Send message
            res = requests.post(
                "https://open.feishu.cn/open-apis/im/v1/messages?receive_id_type=open_id",
                headers={"Authorization": f"Bearer {token}", "Content-Type": "application/json"},
                json={
                    "receive_id": OPEN_ID,
                    "msg_type": "text",
                    "content": json.dumps({"text": text})
                },
                timeout=10
            )
            res.raise_for_status()
            print(f"Feishu push succeeded: {res.json()}")
            return
        except Exception as e:
            print(f"Feishu push attempt {attempt+1} failed: {e}")
            if attempt < max_retries - 1:
                time.sleep(2)
            else:
                print("All Feishu push attempts failed.")

if __name__ == "__main__":
    # Allow running directly with a file path argument (original behavior)
    if len(sys.argv) > 1:
        file_path = sys.argv[1]
        with open(file_path, "r", encoding="utf-8") as f:
            text = f.read()
        push_text(text)
    else:
        print("Usage: python feishu_push.py <file_path>")

def compute_returns(trades, positions):
    """
    Compute total return based on trades and current positions.
    Simple: sum of (sell proceeds - buy cost) + current market value (estimated).
    For simplicity, we assume current price = last execution price.
    """
    total_cost = 0.0
    total_proceeds = 0.0
    for trade in trades:
        qty = float(trade["quantity"])
        price = float(trade["execution_price"])
        if trade["action"] == "buy":
            total_cost += qty * price
        elif trade["action"] == "sell":
            total_proceeds += qty * price

    # Current holdings value (using last known price)
    current_value = 0.0
    for ticker, pos in positions.items():
        # Use the last trade price for that ticker (if available)
        last_price = None
        for trade in reversed(trades):
            if trade["ticker"] == ticker:
                last_price = float(trade["execution_price"])
                break
        if last_price is None:
            last_price = pos["avg_cost"]  # fallback
        current_value += pos["quantity"] * last_price

    total_return = total_proceeds + current_value - total_cost
    if total_cost > 0:
        return_pct = (total_return / total_cost) * 100
    else:
        return_pct = 0.0
    return total_return, return_pct

def compute_max_drawdown(values):
    """
    Compute maximum drawdown from portfolio_value.csv.
    values is list of dicts with 'value' key.

    if not values:
        return 0.0
    peak = float(values[0]["value"])
    max_dd = 0.0
    for row in values:
        v = float(row["value"])
        if v > peak:
            peak = v
        dd = (peak - v) / peak * 100
        if dd > max_dd:
            max_dd = dd
    return max_dd

def parse_report(report_path):
    """
    Parse the weekly report markdown file and extract:
    - position suggestion (percentage)
    - style (growth/value/dividend)
    - top three industries (list of names)
    Returns a dict or None on failure.
    """
    if not os.path.exists(report_path):
        log(f"Report file not found: {report_path}")
        return None

    with open(report_path, "r", encoding="utf-8") as f:
        content = f.read()

    # Extract position percentage
    # Look for "**建议仓位：XX%**"
    pos_match = re.search(r"\*\*建议仓位：(\d+)%\*\*", content)
    if not pos_match:
        log("Could not find position suggestion in report.")
        return None
    position = int(pos_match.group(1))

    # Extract style
    style_match = re.search(r"\*\*风格：(.+?)\*\*", content)
    if not style_match:
        log("Could not find style in report.")
        return None
    style = style_match.group(1).strip()

    # Extract top three industries (lines starting with "1. **...**")
    industry_matches = re.findall(r"\d+\. \*\*(.+?)\*\*", content)
    if len(industry_matches) < 3:
        log("Could not find three industries in report.")
        return None
    industries = industry_matches[:3]

    return {
        "position": position,
        "style": style,
        "industries": industries
    }

def get_next_trading_day():
    """
    Return the next trading day as a date string (YYYY-MM-DD).
    For simplicity, we assume weekdays are trading days.
    If today is Friday, next trading day is Monday.
    """
    today = date.today()
    # Move to next weekday
    next_day = today + timedelta(days=1)
    while next_day.weekday() >= 5:  # Saturday=5, Sunday=6
        next_day += timedelta(days=1)
    return next_day.isoformat()
